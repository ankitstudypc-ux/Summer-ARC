function getDayType(d) {
  const dow = d.getDay();
  if(dow===0) return 'sun';
  if(dow===6) return 'sat';
  if(dow===1||dow===3||dow===5) return 'tuition';
  return 'free';
}

function getMinutes(d){return d.getHours()*60+d.getMinutes();}

function getCurrentBlock(schedule, mins) {
  for(let i=0;i<schedule.length;i++){
    const b=schedule[i];
    const next=schedule[i+1];
    if(mins>=b.s && (next?mins<next.s:true)) return {block:b,nextBlock:next||null,idx:i};
  }
  return {block:schedule[0],nextBlock:schedule[1],idx:0};
}

function formatTime(mins) {
  const h=Math.floor(mins/60)%24;
  const m=mins%60;
  const ampm=h>=12?'PM':'AM';
  const hh=h>12?h-12:(h===0?12:h);
  return `${hh}:${String(m).padStart(2,'0')} ${ampm}`;
}

function initWeeklyPage() {
  const now=new Date();
  const todayDow=now.getDay();
  const grid=document.getElementById('weekly-grid');
  grid.innerHTML='';
  WEEKLY_DATA.forEach((d,i)=>{
    const isToday=i===todayDow;
    const card=document.createElement('div');
    card.className='wday-card'+(isToday?' today-wday':'');
    const blocksHtml=d.blocks.map(b=>{
      const c=TYPE_COLORS[b.type]||TYPE_COLORS.rest;
      return `<div class="wday-block" style="background:${c.bg};color:${c.text};border:1px solid ${c.dot}22">${b.label}</div>`;
    }).join('');
    card.innerHTML=`
      <div class="wday-header" style="${isToday?'color:var(--gold)':''}">
        <span>${d.day}</span>
        ${isToday?'<div class="wday-today-dot"></div>':''}
      </div>
      <div class="wday-body">
        <div style="font-size:10px;color:var(--text2);margin-bottom:6px">${d.role}</div>
        ${blocksHtml}
      </div>
    `;
    grid.appendChild(card);
  });
}

function loadTopicStates(){try{return JSON.parse(localStorage.getItem('sv_topics')||'{}')}catch(e){return{}}}
function saveTopicStates(state){try{localStorage.setItem('sv_topics',JSON.stringify(state))}catch(e){}}

function initTopicsPage() {
  const state=loadTopicStates();
  const container=document.getElementById('topics-content');
  container.innerHTML='';
  
  Object.entries(TOPICS).forEach(([key,subj])=>{
    const done=subj.topics.filter(t=>state[t.id]).length;
    const total=subj.topics.length;
    const pct=Math.round(done/total*100);
    
    const sec=document.createElement('div');
    sec.className='subject-section';
    sec.innerHTML=`
      <div class="subject-header">
        <div class="subject-icon" style="background:${subj.bg};color:${subj.color};font-size:16px">${subj.icon}</div>
        <div class="subject-title" style="color:${subj.color}">${subj.label}</div>
        <div class="subject-count" id="count-${key}">${done}/${total}</div>
        <div class="subject-progress-bar"><div class="subject-progress-fill" style="width:${pct}%;background:${subj.color}" id="bar-${key}"></div></div>
      </div>
      <div class="topics-grid" id="grid-${key}"></div>
    `;
    container.appendChild(sec);
    
    const grid=document.getElementById('grid-'+key);
    subj.topics.forEach(t=>{
      const isDone=!!state[t.id];
      const card=document.createElement('div');
      card.className='topic-card'+(isDone?' done':'');
      card.id='tc-'+t.id;
      const tagClass=t.tag==='must'?'tag-must':t.tag==='strong'?'tag-strong':'tag-bonus';
      const tagLabel=t.tag==='must'?'Must Cover':t.tag==='strong'?'Gives Edge':'Bonus';
      card.innerHTML=`
        <div class="tc-check ${isDone?'checked':''}" id="chk-${t.id}"></div>
        <div class="tc-body">
          <div class="tc-name">${t.name}</div>
          <span class="tc-tag ${tagClass}">${tagLabel}</span>
        </div>
      `;
      card.addEventListener('click',()=>toggleTopic(t.id,key,subj));
      grid.appendChild(card);
    });
  });
}

function toggleTopic(id, key, subj) {
  const state=loadTopicStates();
  state[id]=!state[id];
  saveTopicStates(state);
  
  const card=document.getElementById('tc-'+id);
  const chk=document.getElementById('chk-'+id);
  if(state[id]){card.classList.add('done');chk.classList.add('checked');}
  else{card.classList.remove('done');chk.classList.remove('checked');}
  
  const done=subj.topics.filter(t=>state[t.id]).length;
  const total=subj.topics.length;
  const pct=Math.round(done/total*100);
  document.getElementById('count-'+key).textContent=`${done}/${total}`;
  document.getElementById('bar-'+key).style.width=pct+'%';
}
